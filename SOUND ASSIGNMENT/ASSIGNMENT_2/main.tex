	\textbf{Problem 2.21(c)} Consider an arbitrary linear system with input x[n] and output y[n]. Show that if x[n] = 0
for all n, then y[n] must also be zero for all n.

	\solution 
	For an arbitrary linear system,
	\begin{align}
	    y[n] = T\cbrak{x[n]},
	\end{align}
	
	Let x[n] = 0 for all n.
	\begin{align}
	    y[n] = T\cbrak{x[n]}
	\end{align}
	
	For some arbitrary $x_1$[n], we have
	\begin{align}
	    y_1[n] = T\cbrak{x_1[n]}
	\end{align}
	
	Using the linearity of the system:
	\begin{align}
	    T\cbrak{x[n] + x_1[n]} &= T\cbrak{x[n]} + T\cbrak{x_1[n]}\\
	    &= y[n] + y_1[n]
	\end{align}
	
	Since $x_1$[n] is zero for all n,
	\begin{align}
	    T\cbrak{x[n] + x_1[n]} &= T\cbrak{x[n]} = y_1[n]
	\end{align}
	
	\\Hence, y[n] must also be zero for all n.
